🔁 STEP 3: SIMULATION ENGINE (engine.js)

This is where agents interact.

import { agentTrend, agentMeanReversion, agentNoise } from "./agents";

export function runStep(state) {
  const trendForce = agentTrend(state);
  const reversionForce = agentMeanReversion(state);
  const noiseForce = agentNoise();

  const totalForce =
    0.5 * trendForce +
    0.3 * reversionForce +
    0.2 * noiseForce;

  const newValue = state.value + totalForce;

  return {
    ...state,
    value: newValue,
    history: [...state.history, newValue],
    trend: newValue - state.value,
  };
}

🧠 STEP 4: SIMULATION HOOK (useSimulation.js)

Controls time evolution.

import { useEffect, useState } from "react";
import { runStep } from "./engine";

export function useSimulation() {
  const [state, setState] = useState({
    value: 100,
    average: 100,
    trend: 0,
    history: [100],
  });

  useEffect(() => {
    const interval = setInterval(() => {
      setState((prev) => runStep(prev));
    }, 500);

    return () => clearInterval(interval);
  }, []);

  return state;
}
